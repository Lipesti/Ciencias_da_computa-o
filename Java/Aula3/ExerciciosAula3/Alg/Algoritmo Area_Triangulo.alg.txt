Algoritmo "Area_Triangulo"
Var
    base, altura, area: real
Inicio
    // Entrada de dados
    Escreva("Informe a base do triângulo: ")
    Leia(base)
    Escreva("Informe a altura do triângulo: ")
    Leia(altura)
    
    // Cálculo da área
    area <- (base * altura) / 2
    
    // Saída de dados
    Escreva("A área do triângulo é: ", area)
Fimalgoritmo
