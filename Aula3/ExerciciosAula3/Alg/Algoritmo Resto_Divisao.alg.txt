Algoritmo "Resto_Divisao"
Var
    num1, num2, resto: inteiro
Inicio
    // Entrada de dados
    Escreva("Informe o primeiro número: ")
    Leia(num1)
    Escreva("Informe o segundo número: ")
    Leia(num2)
    
    // Cálculo do resto
    resto <- num1 % num2
    
    // Saída de dados
    Escreva("O resto da divisão de ", num1, " por ", num2, " é: ", resto)
Fimalgoritmo
